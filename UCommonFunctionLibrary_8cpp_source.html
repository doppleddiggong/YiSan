87UMaterialInstanceDynamic* UCommonFunctionLibrary::GetOrCreateMID(
88 UPrimitiveComponent* Target, int32 ElementIndex, FName OptionalName)
89{
90 if (!Target)
91 return nullptr;
92
93 const int32 Num = Target->GetNumMaterials();
94 if (ElementIndex < 0 || ElementIndex >= Num)
95 return nullptr;
96
97 UMaterialInterface* CurMat = Target->GetMaterial(ElementIndex);
98 if (!CurMat)
99 return nullptr;
100
101 if (UMaterialInstanceDynamic* AsMID = Cast<UMaterialInstanceDynamic>(CurMat))
102 return AsMID;
103
104 UMaterialInstanceDynamic* NewMID = UMaterialInstanceDynamic::Create(CurMat, Target, OptionalName);
105 if (NewMID)
106 Target->SetMaterial(ElementIndex, NewMID);
107
108 return NewMID;
109}

111void UCommonFunctionLibrary::PlayLocationSound(const AActor* Actor, USoundBase* Sound, const float RetriggerDelay)
112{
113 if (!IsValid(Actor) || !Sound)
114 return;
115
116 if (auto World = Actor->GetWorld())
117 {
118 if ( RetriggerDelay > 0.0f)
119 {
120 TWeakObjectPtr<const AActor> WeakActor = Actor;
121 FTimerHandle TimerHandle;
122 World->GetTimerManager().SetTimer(TimerHandle, [WeakActor, Sound]()
123 {
124 if (WeakActor.IsValid())
125 {
126 UGameplayStatics::PlaySoundAtLocation(WeakActor.Get(), Sound, WeakActor->GetActorLocation());
127 }
128 }, RetriggerDelay, false);
129 }
130 else
131 {
132 UGameplayStatics::PlaySoundAtLocation(Actor, Sound, Actor->GetActorLocation());
133 }
134 }
135}
