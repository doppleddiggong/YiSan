59void UGoogleSpeechService::RequestSpeechToText(const FString& CorrelationId, const TArray<uint8>& AudioPayload, int32 SampleRate, int32 NumChannels, const FGoogleSpeechToTextDelegate& Delegate)
60{
61 if (ApiKey.IsEmpty())
62 {
63 NETWORK_LOG(TEXT("[GoogleSTT][FAIL] CorrelationId=%s ApiKey missing."), *CorrelationId);
64 Delegate.ExecuteIfBound(false, CorrelationId, FString(), 0.0f);
65 return;
66 }
67
68 const FString Url = BuildSpeechToTextUrl();
69
70 TSharedPtr<FJsonObject> Config = MakeShared<FJsonObject>();
71 Config->SetStringField(TEXT("languageCode"), LanguageCode);
72 Config->SetStringField(TEXT("encoding"), TEXT("LINEAR16"));
73 Config->SetNumberField(TEXT("sampleRateHertz"), SampleRate > 0 ? SampleRate : DefaultSpeechSampleRate);
74 Config->SetNumberField(TEXT("audioChannelCount"), NumChannels > 0 ? NumChannels : 1);
75
76 TSharedPtr<FJsonObject> Audio = MakeShared<FJsonObject>();
77 Audio->SetStringField(TEXT("content"), FBase64::Encode(AudioPayload));
78
79 TSharedPtr<FJsonObject> Root = MakeShared<FJsonObject>();
80 Root->SetObjectField(TEXT("config"), Config);
81 Root->SetObjectField(TEXT("audio"), Audio);
82
83 FString JsonPayload;
84 TSharedRef<TJsonWriter<>> Writer = TJsonWriterFactory<>::Create(&JsonPayload);
85 FJsonSerializer::Serialize(Root.ToSharedRef(), Writer);
86
87 TSharedRef<IHttpRequest, ESPMode::ThreadSafe> Request = FHttpModule::Get().CreateRequest();
88 Request->SetURL(Url);
89 Request->SetVerb(TEXT("POST"));
90 Request->SetHeader(TEXT("Content-Type"), TEXT("application/json"));
91 Request->SetContentAsString(JsonPayload);
92 Request->OnProcessRequestComplete().BindUObject(this, &UGoogleSpeechService::HandleSpeechToTextResponse);
93
94 FPendingSpeechToTextRequest& Pending = PendingSpeechToText.FindOrAdd(&*Request);
95 Pending.CorrelationId = CorrelationId;
96 Pending.Delegate = Delegate;
97 Pending.SampleRate = SampleRate;
98 Pending.NumChannels = NumChannels;
99
100 NETWORK_LOG(TEXT("[REQ][GoogleSTT] CorrelationId=%s PayloadBytes=%d"), *CorrelationId, AudioPayload.Num());
101 Request->ProcessRequest();
102}

104void UGoogleSpeechService::RequestTextToSpeech(const FString& CorrelationId, const FString& Text, int32 SampleRate, const FGoogleTextToSpeechDelegate& Delegate)
105{
106 if (ApiKey.IsEmpty())
107 {
108 NETWORK_LOG(TEXT("[GoogleTTS][FAIL] CorrelationId=%s ApiKey missing."), *CorrelationId);
109 Delegate.ExecuteIfBound(false, CorrelationId, Text, TArray<uint8>(), 0);
110 return;
111 }
112
113 const FString Url = BuildTextToSpeechUrl();
114
115 TSharedPtr<FJsonObject> Input = MakeShared<FJsonObject>();
116 Input->SetStringField(TEXT("text"), Text);
117
118 TSharedPtr<FJsonObject> Voice = MakeShared<FJsonObject>();
119 Voice->SetStringField(TEXT("languageCode"), LanguageCode);
120 Voice->SetStringField(TEXT("name"), VoiceName);
121 if (!VoiceGender.IsEmpty())
122 {
123 Voice->SetStringField(TEXT("ssmlGender"), VoiceGender);
124 }
125
126 TSharedPtr<FJsonObject> AudioConfig = MakeShared<FJsonObject>();
127 AudioConfig->SetStringField(TEXT("audioEncoding"), AudioEncoding);
128 AudioConfig->SetNumberField(TEXT("sampleRateHertz"), SampleRate > 0 ? SampleRate : DefaultTtsSampleRate);
129
130 TSharedPtr<FJsonObject> Root = MakeShared<FJsonObject>();
131 Root->SetObjectField(TEXT("input"), Input);
132 Root->SetObjectField(TEXT("voice"), Voice);
133 Root->SetObjectField(TEXT("audioConfig"), AudioConfig);
134
135 FString JsonPayload;
136 TSharedRef<TJsonWriter<>> Writer = TJsonWriterFactory<>::Create(&JsonPayload);
137 FJsonSerializer::Serialize(Root.ToSharedRef(), Writer);
138
139 TSharedRef<IHttpRequest, ESPMode::ThreadSafe> Request = FHttpModule::Get().CreateRequest();
140 Request->SetURL(Url);
141 Request->SetVerb(TEXT("POST"));
142 Request->SetHeader(TEXT("Content-Type"), TEXT("application/json"));
143 Request->SetContentAsString(JsonPayload);
144 Request->OnProcessRequestComplete().BindUObject(this, &UGoogleSpeechService::HandleTextToSpeechResponse);
145
146 FPendingTextToSpeechRequest& Pending = PendingTextToSpeech.FindOrAdd(&*Request);
147 Pending.CorrelationId = CorrelationId;
148 Pending.Delegate = Delegate;
149 Pending.SampleRate = SampleRate;
150 Pending.Text = Text;
151
152 NETWORK_LOG(TEXT("[REQ][GoogleTTS] CorrelationId=%s TextBytes=%d"), *CorrelationId, Text.Len());
153 Request->ProcessRequest();
154}

156void UGoogleSpeechService::HandleSpeechToTextResponse(FHttpRequestPtr Request, FHttpResponsePtr Response, bool bWasSuccessful)
157{
158 FPendingSpeechToTextRequest Pending;
159 if (!PendingSpeechToText.RemoveAndCopyValue(Request.Get(), Pending))
160 {
161 return;
162 }
163
164 const FString CorrelationId = Pending.CorrelationId;
165
166 if (!bWasSuccessful || !Response.IsValid())
167 {
168 NETWORK_LOG(TEXT("[RES][GoogleSTT][FAIL] CorrelationId=%s Request failed (transport)."), *CorrelationId);
169 Pending.Delegate.ExecuteIfBound(false, CorrelationId, FString(), 0.0f);
170 return;
171 }
172
173 const int32 StatusCode = Response->GetResponseCode();
174 NETWORK_LOG(TEXT("[RES][GoogleSTT] CorrelationId=%s Code=%d"), *CorrelationId, StatusCode);
175
176 if (StatusCode != 200)
177 {
178 Pending.Delegate.ExecuteIfBound(false, CorrelationId, FString(), 0.0f);
179 return;
180 }
181
182 FString Transcript;
183 float Confidence = 0.0f;
184 bool bParsed = false;
185
186 const FString Body = Response->GetContentAsString();
187 TSharedPtr<FJsonObject> Root;
188 const TSharedRef<TJsonReader<>> Reader = TJsonReaderFactory<>::Create(Body);
189 if (FJsonSerializer::Deserialize(Reader, Root) && Root.IsValid())
190 {
191 const TArray<TSharedPtr<FJsonValue>>* Results = nullptr;
192 if (Root->TryGetArrayField(TEXT("results"), Results) && Results && Results->Num() > 0)
193 {
194 const TSharedPtr<FJsonObject> FirstResult = (*Results)[0]->AsObject();
195 const TArray<TSharedPtr<FJsonValue>>* Alternatives = nullptr;
196 if (FirstResult.IsValid() && FirstResult->TryGetArrayField(TEXT("alternatives"), Alternatives) && Alternatives && Alternatives->Num() > 0)
197 {
198 const TSharedPtr<FJsonObject> FirstAlt = (*Alternatives)[0]->AsObject();
199 if (FirstAlt.IsValid())
200 {
201 FirstAlt->TryGetStringField(TEXT("transcript"), Transcript);
202 double ConfidenceDouble = 0.0;
203 if (FirstAlt->TryGetNumberField(TEXT("confidence"), ConfidenceDouble))
204 {
205 Confidence = static_cast<float>(ConfidenceDouble);
206 }
207 bParsed = !Transcript.IsEmpty();
208 }
209 }
210 }
211 }
212
213 Pending.Delegate.ExecuteIfBound(bParsed, CorrelationId, Transcript, Confidence);
214}

216void UGoogleSpeechService::HandleTextToSpeechResponse(FHttpRequestPtr Request, FHttpResponsePtr Response, bool bWasSuccessful)
217{
218 FPendingTextToSpeechRequest Pending;
219 if (!PendingTextToSpeech.RemoveAndCopyValue(Request.Get(), Pending))
220 {
221 return;
222 }
223
224 const FString CorrelationId = Pending.CorrelationId;
225
226 if (!bWasSuccessful || !Response.IsValid())
227 {
228 NETWORK_LOG(TEXT("[RES][GoogleTTS][FAIL] CorrelationId=%s Request failed (transport)."), *CorrelationId);
229 Pending.Delegate.ExecuteIfBound(false, CorrelationId, Pending.Text, TArray<uint8>(), 0);
230 return;
231 }
232
233 const int32 StatusCode = Response->GetResponseCode();
234 NETWORK_LOG(TEXT("[RES][GoogleTTS] CorrelationId=%s Code=%d"), *CorrelationId, StatusCode);
235
236 if (StatusCode != 200)
237 {
238 Pending.Delegate.ExecuteIfBound(false, CorrelationId, Pending.Text, TArray<uint8>(), 0);
239 return;
240 }
241
242 TArray<uint8> AudioData;
243 bool bParsed = false;
244
245 const FString Body = Response->GetContentAsString();
246 TSharedPtr<FJsonObject> Root;
247 const TSharedRef<TJsonReader<>> Reader = TJsonReaderFactory<>::Create(Body);
248 if (FJsonSerializer::Deserialize(Reader, Root) && Root.IsValid())
249 {
250 FString AudioContent;
251 if (Root->TryGetStringField(TEXT("audioContent"), AudioContent))
252 {
253 bParsed = FBase64::Decode(AudioContent, AudioData);
254 }
255 }
256
257 Pending.Delegate.ExecuteIfBound(bParsed, CorrelationId, Pending.Text, AudioData, Pending.SampleRate > 0 ? Pending.SampleRate : DefaultTtsSampleRate);
258}
