28USTRUCT(BlueprintType)
29struct FParabolaBallisticTrack
30{
31 GENERATED_BODY()
32
33 // 입력 파라미터
34 UPROPERTY(EditAnywhere, BlueprintReadWrite) FVector Start = FVector::ZeroVector;
35 UPROPERTY(EditAnywhere, BlueprintReadWrite, meta=(ClampMin="0.001")) float Duration = 1.0f;
36 UPROPERTY(EditAnywhere, BlueprintReadWrite) float GravityZ = -980.f; // UE 기본 cm/s^2
37 UPROPERTY(EditAnywhere, BlueprintReadWrite) FVector Direction = FVector(1,0,0); // 자동 정규화
38 UPROPERTY(EditAnywhere, BlueprintReadWrite, meta=(ClampMin="0.0")) float Power = 1200.f; // cm/s
39
40 // 평면 고정(선택)
41 UPROPERTY(EditAnywhere, BlueprintReadWrite) bool bLockZ = false;
42 UPROPERTY(EditAnywhere, BlueprintReadWrite, meta=(EditCondition="bLockZ")) float LockedZ = 0.f;
43
44 // 런타임 상태
45 UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Transient) float CurrentTime = 0.f;
46
47 // 타임 유틸
49 FORCEINLINE void ResetTime() { CurrentTime = 0.f; }
51 FORCEINLINE float GetAlpha() const { return FMath::Clamp(CurrentTime / FMath::Max(0.001f, Duration), 0.f, 1.f); }
53 FORCEINLINE void Advance(float DeltaTime)
54 {
55 CurrentTime = FMath::Min(CurrentTime + FMath::Max(0.f, DeltaTime), FMath::Max(0.001f, Duration));
56 }
57
58 // 속도/평가
60 FORCEINLINE FVector GetInitialVelocity() const
61 {
62 return Direction.GetSafeNormal() * Power;
63 }
64
66 FVector EvaluateAtTime(const AActor* /*Owner*/, float TimeSec) const
67 {
68 const float t = FMath::Clamp(TimeSec, 0.f, FMath::Max(0.001f, Duration));
69 const FVector g(0,0,GravityZ);
70 const FVector V0 = GetInitialVelocity();
71
72 FVector P = Start + V0 * t + 0.5f * g * t * t;
73 if (bLockZ)
74 P.Z = LockedZ;
75 return P;
76 }
77
79 FORCEINLINE FVector EvaluateAtAlpha(const AActor* Owner, float Alpha) const
80 {
81 return EvaluateAtTime(Owner, Alpha * FMath::Max(0.001f, Duration));
82 }
83
85 FORCEINLINE FVector EvaluateAtCurrent(const AActor* Owner) const
86 {
87 return EvaluateAtTime(Owner, CurrentTime);
88 }
89
90 // 목표점+시간으로 V0 역산 → Direction/Power 세팅
92 static FVector SolveV0ForArc(const FVector& InStart, const FVector& InEnd, float t, float InGravityZ=-980.f)
93 {
94 return FMathHelper::SolveV0ForProjectile( InStart, InEnd, t, InGravityZ );
95 }
96
98 static void SplitVelocity(const FVector& V0, FVector& OutDir, float& OutPow)
99 {
100 OutPow = V0.Length();
101 OutDir = (OutPow > SMALL_NUMBER) ? V0 / OutPow : FVector(1,0,0);
102 }
103
105 void ApplyArcSolution(const FVector& InStart, const FVector& InEnd, float t)
106 {
107 Start = InStart;
108 Duration = FMath::Max(0.001f, t);
109 const FVector V0 = SolveV0ForArc(InStart, InEnd, Duration, GravityZ);
110 SplitVelocity(V0, Direction, Power);
111 ResetTime();
112 }
113};

116USTRUCT(BlueprintType)
117struct FParabolaGeometricTrack
118{
119 GENERATED_BODY()
120
121 // 입력 파라미터
122 UPROPERTY(EditAnywhere, BlueprintReadWrite) FVector Start = FVector::ZeroVector;
123 UPROPERTY(EditAnywhere, BlueprintReadWrite) FVector Target = FVector::ZeroVector;
124 UPROPERTY(EditAnywhere, BlueprintReadWrite, meta=(ClampMin="0.001")) float Duration = 1.0f;
125 UPROPERTY(EditAnywhere, BlueprintReadWrite, meta=(ClampMin="0.0")) float ApexHeight = 200.f; // 정점에서 +H
126 UPROPERTY(EditAnywhere, BlueprintReadWrite) FVector UpAxis = FVector::UpVector;
127
128 // 평면 고정(선택)
129 UPROPERTY(EditAnywhere, BlueprintReadWrite) bool bLockZ = false;
130 UPROPERTY(EditAnywhere, BlueprintReadWrite, meta=(EditCondition="bLockZ")) float LockedZ = 0.f;
131
132 // 런타임 상태
133 UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Transient) float CurrentTime = 0.f;
134
135 // 타임 유틸
137 FORCEINLINE void ResetTime() { CurrentTime = 0.f; }
139 FORCEINLINE float GetAlpha() const { return FMath::Clamp(CurrentTime / FMath::Max(0.001f, Duration), 0.f, 1.f); }
141 FORCEINLINE void Advance(float DeltaTime)
142 {
143 CurrentTime = FMath::Min(CurrentTime + FMath::Max(0.f, DeltaTime), FMath::Max(0.001f, Duration));
144 }
145
146 // 평가(직선보간 + 4Ha(1-a) 버프)
148 FVector EvaluateAtTime(const AActor* /*Owner*/, float TimeSec) const
149 {
150 const float D = FMath::Max(0.001f, Duration);
151 const float t = FMath::Clamp(TimeSec, 0.f, D);
152 const float a = t / D;
153
154 return FMathHelper::InterpArcSin(Start, Target, ApexHeight, a);
155 }
156
158 FORCEINLINE FVector EvaluateAtAlpha(const AActor* Owner, float Alpha) const
159 {
160 return EvaluateAtTime(Owner, Alpha * FMath::Max(0.001f, Duration));
161 }
162
164 FORCEINLINE FVector EvaluateAtCurrent(const AActor* Owner) const
165 {
166 return EvaluateAtTime(Owner, CurrentTime);
167 }
168
170 void ApplyArcSolution(const FVector& InStart, const FVector& InTarget, const float InApexHeight, float t, const FVector& InUpAxis = FVector::UpVector)
171 {
172 Start = InStart;
173 Target = InTarget;
174 ApexHeight = FMath::Max(0.f, InApexHeight);
175 UpAxis = InUpAxis;
176 Duration = FMath::Max(0.001f, t);
177 ResetTime();
178 }
179};
