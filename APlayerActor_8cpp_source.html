71void APlayerActor::Cmd_Move_Implementation(const FVector2D& Axis)
72{
73 if (!IsMoveEnable() || !Controller)
74 {
75 return;
76 }
77
78 // Use the controller's rotation to determine movement direction.
79 const FRotator ControlRotation = Controller->GetControlRotation();
80 const EMovementMode CurrentMovementMode = GetCharacterMovement()->MovementMode;
81
82 if (CurrentMovementMode == MOVE_Walking || CurrentMovementMode == MOVE_Falling)
83 {
84 // For ground movement, only use the Yaw rotation to prevent pitching into the ground.
85 const FRotator YawRotation(0.0f, ControlRotation.Yaw, 0.0f);
86
87 // Calculate forward and right vectors based on the Yaw rotation.
88 const FVector ForwardDirection = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::X);
89 const FVector RightDirection = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::Y);
90
91 AddMovementInput(ForwardDirection, Axis.Y);
92 AddMovementInput(RightDirection, Axis.X);
93 }
94 else if (CurrentMovementMode == MOVE_Flying)
95 {
96 // For flying, use the full 3D rotation of the controller.
97 const FRotator FullRotation(ControlRotation.Pitch, ControlRotation.Yaw, 0.0f);
98
99 const FVector ForwardDirection = FRotationMatrix(FullRotation).GetUnitAxis(EAxis::X);
100 const FVector RightDirection = FRotationMatrix(FullRotation).GetUnitAxis(EAxis::Y);
101
102 AddMovementInput(ForwardDirection, Axis.Y);
103 AddMovementInput(RightDirection, Axis.X);
104 }
105}
