68void APlayerActor::Cmd_Move_Implementation(const FVector2D& Axis)
69{
70 if ( !Controller)
71 return;
72
73 const FRotator ControlRotation = Controller->GetControlRotation();
74 const FRotator YawRotation(0.0f, ControlRotation.Yaw, 0.0f);
75
76 const FVector ForwardDirection = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::X);
77 const FVector RightDirection = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::Y);
78
79 AddMovementInput(ForwardDirection, Axis.Y);
80 AddMovementInput(RightDirection, Axis.X);
81}
