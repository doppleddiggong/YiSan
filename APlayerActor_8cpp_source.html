102void APlayerActor::Cmd_Move_Implementation(const FVector2D& Axis)
103{
104 if (!IsMoveEnable() || !Controller)
105 {
106 return;
107 }
108
109 // Use the controller's rotation to determine movement direction.
110 const FRotator ControlRotation = Controller->GetControlRotation();
111 const EMovementMode CurrentMovementMode = GetCharacterMovement()->MovementMode;
112
113 if (CurrentMovementMode == MOVE_Walking || CurrentMovementMode == MOVE_Falling)
114 {
115 // For ground movement, only use the Yaw rotation to prevent pitching into the ground.
116 const FRotator YawRotation(0.0f, ControlRotation.Yaw, 0.0f);
117
118 // Calculate forward and right vectors based on the Yaw rotation.
119 const FVector ForwardDirection = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::X);
120 const FVector RightDirection = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::Y);
121
122 AddMovementInput(ForwardDirection, Axis.Y);
123 AddMovementInput(RightDirection, Axis.X);
124 }
125 else if (CurrentMovementMode == MOVE_Flying)
126 {
127 // For flying, use the full 3D rotation of the controller.
128 const FRotator FullRotation(ControlRotation.Pitch, ControlRotation.Yaw, 0.0f);
129
130 const FVector ForwardDirection = FRotationMatrix(FullRotation).GetUnitAxis(EAxis::X);
131 const FVector RightDirection = FRotationMatrix(FullRotation).GetUnitAxis(EAxis::Y);
132
133 AddMovementInput(ForwardDirection, Axis.Y);
134 AddMovementInput(RightDirection, Axis.X);
135 }
136}
