57void UGameDataManager::LoadData_HitStopTable()
58{
59 HitStopCache.Reset();
60 bLoadHitStop = false;
61
62 UDataTable* TableObj = HitStopTable.LoadSynchronous();
63 if (!TableObj)
64 {
65 PRINTLOG(TEXT("Load failed: %s"), *HitStopTable.ToString());
66 return;
67 }
68
69 static const FString ContextString(TEXT("HitStopTable"));
70 for (const FName& RowName : TableObj->GetRowNames() )
71 {
72 if (const FHitStopData* Row = TableObj->FindRow<FHitStopData>(RowName, ContextString, true))
73 {
74 HitStopCache.Add(Row->Type, *Row);
75 }
76 }
77
78 bLoadHitStop = true;
79}
