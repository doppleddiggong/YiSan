29void UYiSanVoiceConsoleSubsystem::HandleSpeakCommand(const TArray<FString>& Args, UWorld* World)
30{
31 if (!World)
32 {
33 UE_LOG(LogYiSan, Warning, TEXT("yisan.tts called without a valid world."));
34 return;
35 }
36
37 if (Args.Num() == 0)
38 {
39 UE_LOG(LogYiSan, Warning, TEXT("yisan.tts requires text input."));
40 return;
41 }
42
43 FString Text = FString::Join(Args, TEXT(" "));
44 Text.TrimStartAndEndInline();
45
46 if (Text.IsEmpty())
47 {
48 UE_LOG(LogYiSan, Warning, TEXT("yisan.tts ignored empty text."));
49 return;
50 }
51
52 for (TActorIterator<APlayerActor> It(World); It; ++It)
53 {
54 if (APlayerActor* Player = *It)
55 {
56 if (UVoiceCaptureComponent* VoiceComponent = Player->FindComponentByClass<UVoiceCaptureComponent>())
57 {
58 VoiceComponent->RequestSynthesis(Text);
59 UE_LOG(LogYiSan, Log, TEXT("yisan.tts dispatched text (%s)."), *Text);
60 return;
61 }
62 }
63 }
64
65 UE_LOG(LogYiSan, Warning, TEXT("yisan.tts could not find an APlayerActor with UVoiceCaptureComponent."));
66}
