60void AGameCharacter::SetupCharacterFromType(const ECharacterType Type, const bool bIsAnother)
61{
62 this->CharacterType = Type;
63
64 FCharacterAssetData Params;
65 UGameDataManager::Get(GetWorld())->GetCharacterAssetData(Type, Params);
66
67 if (Params.CharacterDataAsset.IsValid())
68 this->CharacterData = Params.CharacterDataAsset.Get();
69 else
70 this->CharacterData = Params.CharacterDataAsset.LoadSynchronous();
71
72 MeshComp = GetMesh();
73 if ( IsValid(MeshComp) )
74 {
75 if (CharacterData->MeshData.IsValid())
76 MeshComp->SetSkeletalMesh(CharacterData->MeshData.Get());
77 else
78 MeshComp->SetSkeletalMesh(CharacterData->MeshData.LoadSynchronous());
79
80 MeshComp->EmptyOverrideMaterials();
81 for (int32 i = 0; i < CharacterData->MaterialArray.Num(); ++i)
82 {
83 UMaterialInstanceConstant* MaterialInst = nullptr;
84 if (CharacterData->MaterialArray[i].IsValid())
85 MaterialInst = CharacterData->MaterialArray[i].Get();
86 else
87 MaterialInst = CharacterData->MaterialArray[i].LoadSynchronous();
88
89 if (MaterialInst)
90 {
91 auto DynamicMat = UMaterialInstanceDynamic::Create(MaterialInst, MeshComp);
92 if (DynamicMat)
93 {
94 // 1번이 Cloth
95 if (i == 1)
96 DynamicMat->SetScalarParameterValue(FName("AnotherSide"), bIsAnother ? 1.0f : 0.0f);
97 MeshComp->SetMaterial(i, DynamicMat);
98 }
99 }
100 }
101
102 MeshComp->SetRelativeLocation( CharacterData->RelativeLocation);
103 MeshComp->SetRelativeRotation( CharacterData->RelativeRotator );
104 MeshComp->SetRelativeScale3D( CharacterData->RelativeScale );
105
106 if (CharacterData->AnimBluePrint.IsValid())
107 MeshComp->SetAnimInstanceClass(CharacterData->AnimBluePrint.Get());
108 else
109 MeshComp->SetAnimInstanceClass(CharacterData->AnimBluePrint.LoadSynchronous());
110
111 auto Rule = FAttachmentTransformRules::SnapToTargetNotIncludingScale;
112
113 LeftHandComp->AttachToComponent(MeshComp, Rule, CharacterData->LeftHandSocketName);
114 RightHandComp->AttachToComponent(MeshComp, Rule, CharacterData->RightHandSocketName);
115 LeftFootComp->AttachToComponent(MeshComp, Rule, CharacterData->LeftFootSocketName);
116 RightFootComp->AttachToComponent(MeshComp, Rule, CharacterData->RightFootSocketName);
117
118 AnimInstance = MeshComp->GetAnimInstance();
119 BindMontageDelegates(AnimInstance);
120 }
121}

125{

199bool AGameCharacter::IsInSight(const AActor* Other) const
200{
201 const FVector SelfLoc = GetActorLocation();
202 const FVector OtherLoc = Other->GetActorLocation();
203 const FVector ToOther = OtherLoc - SelfLoc;
204
205 const float Dist = ToOther.Size();
206 if (Dist > SightRange)
207 return false;
208
209 const FVector Fwd = GetActorForwardVector();
210 const float CosHalfFOV = FMath::Cos(FMath::DegreesToRadians(SightHalfFOVDeg));
211 const float CosAngle = FVector::DotProduct(Fwd, ToOther.GetSafeNormal());
212 if (CosAngle < CosHalfFOV)
213 return false;
214
215 FHitResult Hit;
216 FCollisionQueryParams Params(SCENE_QUERY_STAT(PlayerSightLOS), false, this);
217
218 const bool bBlocked = GetWorld()->LineTraceSingleByChannel(
219 Hit,
220 SelfLoc + FVector(0,0,50),
221 OtherLoc + FVector(0,0,50),
222 ECC_Visibility,
223 Params
224 );
225
226 if (bBlocked && Hit.GetActor() != Other)
227 return false;
228
229 return true;
230}

271{

274
275 auto Movement = this->GetCharacterMovement();

280 this->bUseControllerRotationYaw = true;
