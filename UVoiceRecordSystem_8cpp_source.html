57void UVoiceRecordSystem::HandleOnCapture(const float* InAudio, int32 InNumFrames, int32 InNumChannels, int32 InSampleRate)
58{
59 LastSampleRate = InSampleRate;
60 LastNumChannels = InNumChannels;
61
62 const int32 SampleCount = InNumFrames * InNumChannels;
63 PCMData.Reserve(PCMData.Num() + SampleCount * sizeof(int16));
64
65 for (int32 i = 0; i < SampleCount; ++i)
66 {
67 float Sample = InAudio[i];
68 Sample = FMath::Clamp(Sample, -1.0f, 1.0f);
69
70 int16 Int16Sample = static_cast<int16>(Sample * 32767.0f);
71 const uint8* SampleBytes = reinterpret_cast<const uint8*>(&Int16Sample);
72
73 PCMData.Append(SampleBytes, sizeof(int16));
74 }
