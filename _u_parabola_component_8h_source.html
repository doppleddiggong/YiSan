24USTRUCT(BlueprintType)
25struct FParabolaBallisticTrack
26{
27 GENERATED_BODY()
28
29 // 입력 파라미터
30 UPROPERTY(EditAnywhere, BlueprintReadWrite) FVector Start = FVector::ZeroVector;
31 UPROPERTY(EditAnywhere, BlueprintReadWrite, meta=(ClampMin="0.001")) float Duration = 1.0f;
32 UPROPERTY(EditAnywhere, BlueprintReadWrite) float GravityZ = -980.f; // UE 기본 cm/s^2
33 UPROPERTY(EditAnywhere, BlueprintReadWrite) FVector Direction = FVector(1,0,0); // 자동 정규화
34 UPROPERTY(EditAnywhere, BlueprintReadWrite, meta=(ClampMin="0.0")) float Power = 1200.f; // cm/s
35
36 // 평면 고정(선택)
37 UPROPERTY(EditAnywhere, BlueprintReadWrite) bool bLockZ = false;
38 UPROPERTY(EditAnywhere, BlueprintReadWrite, meta=(EditCondition="bLockZ")) float LockedZ = 0.f;
39
40 // 런타임 상태
41 UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Transient) float CurrentTime = 0.f;
42
43 // 타임 유틸
45 FORCEINLINE void ResetTime() { CurrentTime = 0.f; }
47 FORCEINLINE float GetAlpha() const { return FMath::Clamp(CurrentTime / FMath::Max(0.001f, Duration), 0.f, 1.f); }
49 FORCEINLINE void Advance(float DeltaTime)
50 {
51 CurrentTime = FMath::Min(CurrentTime + FMath::Max(0.f, DeltaTime), FMath::Max(0.001f, Duration));
52 }
53
54 // 속도/평가
56 FORCEINLINE FVector GetInitialVelocity() const
57 {
58 return Direction.GetSafeNormal() * Power;
59 }
60
62 FVector EvaluateAtTime(const AActor* /*Owner*/, float TimeSec) const
63 {
64 const float t = FMath::Clamp(TimeSec, 0.f, FMath::Max(0.001f, Duration));
65 const FVector g(0,0,GravityZ);
66 const FVector V0 = GetInitialVelocity();
67
68 FVector P = Start + V0 * t + 0.5f * g * t * t;
69 if (bLockZ)
70 P.Z = LockedZ;
71 return P;
72 }
73
75 FORCEINLINE FVector EvaluateAtAlpha(const AActor* Owner, float Alpha) const
76 {
77 return EvaluateAtTime(Owner, Alpha * FMath::Max(0.001f, Duration));
78 }
79
81 FORCEINLINE FVector EvaluateAtCurrent(const AActor* Owner) const
82 {
83 return EvaluateAtTime(Owner, CurrentTime);
84 }
85
86 // 목표점+시간으로 V0 역산 → Direction/Power 세팅
88 static FVector SolveV0ForArc(const FVector& InStart, const FVector& InEnd, float t, float InGravityZ=-980.f)
89 {
90 return FMathHelper::SolveV0ForProjectile( InStart, InEnd, t, InGravityZ );
91 }
92
94 static void SplitVelocity(const FVector& V0, FVector& OutDir, float& OutPow)
95 {
96 OutPow = V0.Length();
97 OutDir = (OutPow > SMALL_NUMBER) ? V0 / OutPow : FVector(1,0,0);
98 }
99
101 void ApplyArcSolution(const FVector& InStart, const FVector& InEnd, float t)
102 {
103 Start = InStart;
104 Duration = FMath::Max(0.001f, t);
105 const FVector V0 = SolveV0ForArc(InStart, InEnd, Duration, GravityZ);
106 SplitVelocity(V0, Direction, Power);
107 ResetTime();
108 }
109};

112USTRUCT(BlueprintType)
113struct FParabolaGeometricTrack
114{
115 GENERATED_BODY()
116
117 // 입력 파라미터
118 UPROPERTY(EditAnywhere, BlueprintReadWrite) FVector Start = FVector::ZeroVector;
119 UPROPERTY(EditAnywhere, BlueprintReadWrite) FVector Target = FVector::ZeroVector;
120 UPROPERTY(EditAnywhere, BlueprintReadWrite, meta=(ClampMin="0.001")) float Duration = 1.0f;
121 UPROPERTY(EditAnywhere, BlueprintReadWrite, meta=(ClampMin="0.0")) float ApexHeight = 200.f; // 정점에서 +H
122 UPROPERTY(EditAnywhere, BlueprintReadWrite) FVector UpAxis = FVector::UpVector;
123
124 // 평면 고정(선택)
125 UPROPERTY(EditAnywhere, BlueprintReadWrite) bool bLockZ = false;
126 UPROPERTY(EditAnywhere, BlueprintReadWrite, meta=(EditCondition="bLockZ")) float LockedZ = 0.f;
127
128 // 런타임 상태
129 UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Transient) float CurrentTime = 0.f;
130
131 // 타임 유틸
133 FORCEINLINE void ResetTime() { CurrentTime = 0.f; }
135 FORCEINLINE float GetAlpha() const { return FMath::Clamp(CurrentTime / FMath::Max(0.001f, Duration), 0.f, 1.f); }
137 FORCEINLINE void Advance(float DeltaTime)
138 {
139 CurrentTime = FMath::Min(CurrentTime + FMath::Max(0.f, DeltaTime), FMath::Max(0.001f, Duration));
140 }
141
142 // 평가(직선보간 + 4Ha(1-a) 버프)
144 FVector EvaluateAtTime(const AActor* /*Owner*/, float TimeSec) const
145 {
146 const float D = FMath::Max(0.001f, Duration);
147 const float t = FMath::Clamp(TimeSec, 0.f, D);
148 const float a = t / D;
149
150 return FMathHelper::InterpArcSin(Start, Target, ApexHeight, a);
151 }
152
154 FORCEINLINE FVector EvaluateAtAlpha(const AActor* Owner, float Alpha) const
155 {
156 return EvaluateAtTime(Owner, Alpha * FMath::Max(0.001f, Duration));
157 }
158
160 FORCEINLINE FVector EvaluateAtCurrent(const AActor* Owner) const
161 {
162 return EvaluateAtTime(Owner, CurrentTime);
163 }
164
166 void ApplyArcSolution(const FVector& InStart, const FVector& InTarget, const float InApexHeight, float t, const FVector& InUpAxis = FVector::UpVector)
167 {
168 Start = InStart;
169 Target = InTarget;
170 ApexHeight = FMath::Max(0.f, InApexHeight);
171 UpAxis = InUpAxis;
172 Duration = FMath::Max(0.001f, t);
173 ResetTime();
174 }
175};
