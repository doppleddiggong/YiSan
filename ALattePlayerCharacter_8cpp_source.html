83void ALattePlayerCharacter::Cmd_Move_Implementation(const FVector2D& Axis)
84{
85 if (!IsMoveEnable() || !Controller)
86 {
87 return;
88 }
89
90 // Use the controller's rotation to determine movement direction.
91 const FRotator ControlRotation = Controller->GetControlRotation();
92 const EMovementMode CurrentMovementMode = GetCharacterMovement()->MovementMode;
93
94 if (CurrentMovementMode == MOVE_Walking || CurrentMovementMode == MOVE_Falling)
95 {
96 // For ground movement, only use the Yaw rotation to prevent pitching into the ground.
97 const FRotator YawRotation(0.0f, ControlRotation.Yaw, 0.0f);
98
99 // Calculate forward and right vectors based on the Yaw rotation.
100 const FVector ForwardDirection = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::X);
101 const FVector RightDirection = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::Y);
102
103 AddMovementInput(ForwardDirection, Axis.Y);
104 AddMovementInput(RightDirection, Axis.X);
105 }
106 else if (CurrentMovementMode == MOVE_Flying)
107 {
108 // For flying, use the full 3D rotation of the controller.
109 const FRotator FullRotation(ControlRotation.Pitch, ControlRotation.Yaw, 0.0f);
110
111 const FVector ForwardDirection = FRotationMatrix(FullRotation).GetUnitAxis(EAxis::X);
112 const FVector RightDirection = FRotationMatrix(FullRotation).GetUnitAxis(EAxis::Y);
113
114 AddMovementInput(ForwardDirection, Axis.Y);
115 AddMovementInput(RightDirection, Axis.X);
116 }
117}
