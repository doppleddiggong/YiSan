78void UGameSoundManager::PlaySound2D(const EGameSoundType Type)
79{
80 if (TObjectPtr<USoundBase>* FoundSound = SoundData.Find(Type))
81 {
82 if (*FoundSound)
83 {
84 // 이미 재생 중인 사운드가 있다면 멈추고 제거
85 if (UAudioComponent* ExistingComp = ActiveSounds.FindRef(Type))
86 {
87 ExistingComp->Stop();
88 ActiveSounds.Remove(Type);
89 }
90
91 // 새로운 사운드 재생 및 저장
92 if (UAudioComponent* NewComp = UGameplayStatics::SpawnSound2D(GetWorld(), *FoundSound))
93 {
94 ActiveSounds.Add(Type, NewComp);
95 }
96 }
97 }
