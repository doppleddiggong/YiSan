57FVector UKnockbackSystem::ComputeKnockDir(const AActor* Target, const AActor* Instigator)
58{
59 FVector Dir = FVector::ZeroVector;
60
61 if (Target && Instigator)
62 Dir = Target->GetActorLocation() - Instigator->GetActorLocation();
63
64 if (!Dir.Normalize())
65 {
66 if (Target)
67 {
68 FVector Fwd = Target->GetActorForwardVector();
69 Dir = (-Fwd).GetSafeNormal();
70 }
71 else
72 {
73 Dir = FVector::ForwardVector;
74 }
75 }
76
77 return Dir;
78}

80void UKnockbackSystem::Knockback(AActor* Target, AActor* Instigator, EDamageType Type, float Resistance)
81{
82 FKnockbackData Params;
83 if (auto DataManager = UGameDataManager::Get(GetWorld()))
84 {
85 if (!DataManager->GetKnockbackData(Type, Params))
86 return;
87 }
88
89 const FVector Dir = ComputeKnockDir(Target, Instigator);
90
91 if (MoveComp->MovementMode != MOVE_Falling)
92 MoveComp->SetMovementMode(MOVE_Falling);
93
94 if (!bFriction)
95 {
96 PrevBrakingFriction = MoveComp->BrakingFrictionFactor;
97 bFriction = true;
98 }
99 MoveComp->BrakingFrictionFactor = FMath::Clamp(Params.BrakingFrictionFactor, 0.f, 1.f);
100
101 const float Power = FMath::Max(0.f, Params.KnockbackPower) * (1.f - Clamp01(Resistance));
102
103 FVector Launch = Dir * Power;
104 Launch.Z += Params.UpPower; // Always apply UpPower for testing
105
106 Owner->LaunchCharacter(Launch, true, true);
107
108 GetWorld()->GetTimerManager().SetTimer(
109 RestoreTimer, this, &UKnockbackSystem::RestoreMovement,
110 FMath::Max(0.f, Params.Duration), false
111 );
112
113 if (Params.bAfterFlying)
114 {
115 GetWorld()->GetTimerManager().ClearTimer(FlyingTimer);
116 GetWorld()->GetTimerManager().SetTimer(
117 FlyingTimer, this, &UKnockbackSystem::EnterFlying,
118 FMath::Max(0.f, Params.FlyingDelay), false
119 );
120 }
121}
