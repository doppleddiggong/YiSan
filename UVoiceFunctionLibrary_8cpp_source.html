8TArray<uint8> UVoiceFunctionLibrary::ConvertPCM2WAV(
9 const TArray<uint8>& InPCMData,
10 const int32 InSampleRate,
11 const int32 InChannel,
12 const int32 InBitsPerSample)
13{
14 TArray<uint8> WavData;
15
16 const int32 ByteRate = InSampleRate * InChannel * InBitsPerSample / 8;
17 const int32 BlockAlign = InChannel * InBitsPerSample / 8;
18 const int32 DataSize = InPCMData.Num();
19 const int32 ChunkSize = 36 + DataSize;
20
21 // RIFF 헤더
22 WavData.Append(reinterpret_cast<const uint8*>("RIFF"), 4);
23 WavData.Append(reinterpret_cast<const uint8*>(&ChunkSize), 4);
24 WavData.Append(reinterpret_cast<const uint8*>("WAVE"), 4);
25
26 // fmt chunk
27 WavData.Append(reinterpret_cast<const uint8*>("fmt "), 4);
28 int32 SubChunk1Size = 16;
29 int16 AudioFormat = 1;
30
31 WavData.Append(reinterpret_cast<const uint8*>(&SubChunk1Size), 4);
32 WavData.Append(reinterpret_cast<const uint8*>(&AudioFormat), 2);
33 WavData.Append(reinterpret_cast<const uint8*>(&InChannel), 2);
34 WavData.Append(reinterpret_cast<const uint8*>(&InSampleRate), 4);
35 WavData.Append(reinterpret_cast<const uint8*>(&ByteRate), 4);
36 WavData.Append(reinterpret_cast<const uint8*>(&BlockAlign), 2);
37 WavData.Append(reinterpret_cast<const uint8*>(&InBitsPerSample), 2);
38 WavData.Append(reinterpret_cast<const uint8*>("data"), 4);
39 WavData.Append(reinterpret_cast<const uint8*>(&DataSize), 4);
40
41 WavData.Append(InPCMData);
42 return WavData;
43}
