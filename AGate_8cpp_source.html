128bool AGate::IsPlayerActor(const AActor* OtherActor) const
129{
130 if ( OtherActor == nullptr)
131 return false;
132
133 if ( OtherActor == this)
134 return false;
135
136 if ( OtherActor->ActorHasTag(GameTags::Player) == false)
137 return false;
138
139 return true;
140}

142void AGate::OnOverlapBegin(UPrimitiveComponent* OverlappedComp, AActor* OtherActor, UPrimitiveComponent* OtherComp, int32 OtherBodyIndex, bool bFromSweep, const FHitResult& SweepResult)
143{
144 if (IsPlayerActor(OtherActor))
145 {
146 if ( auto Pawn = Cast<APawn>(OtherActor) )
147 {
148 if ( OverlappingPawns.Contains(Pawn) == false )
149 OverlappingPawns.Add(Pawn);
150 }
151
152 OpenDoor();
153 }
154}
155
156void AGate::OnOverlapEnd(UPrimitiveComponent* OverlappedComp, AActor* OtherActor, UPrimitiveComponent* OtherComp, int32 OtherBodyIndex)
157{
158 if (IsPlayerActor(OtherActor))
159 {
160 if ( auto Pawn = Cast<APawn>(OtherActor) )
161 OverlappingPawns.Remove(Pawn);
162
163 if (OverlappingPawns.Num() == 0)
164 CloseDoor();
165 }
166}
