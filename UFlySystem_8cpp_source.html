81{
82 JumpCount++;

84 return;
85
86 switch (JumpCount)
87 {
88 case 1:
89 {
90 Owner->Jump();
91 }
92 break;
93

95 {
96 this->ActivateUpstream();
97 Owner->SetFlying();
98 }
99 break;
100
101 case 3:
102 {
103 FHitResult HitResult;
104 this->OnLand(HitResult);
105 }
106 break;
107 }

157void UFlySystem::UpstreamTick(const float DeltaTime)
158{

160 return;
161
162 ElapsedTime += DeltaTime;
163
164 FVector ResultLocation = UEaseFunctionLibrary::LerpVectorEase(

167 EEaseType::EaseOutQuart);
168
169 Owner->SetActorLocation(ResultLocation, true);
170
171 float Dist = FVector::Dist( Owner->GetActorLocation(), EndLocation );
172 if ( Dist < AlmostDist || ElapsedTime > UpstreamDuration )
173 {
174 bIsUpstream = false;

203FVector UFlySystem::AltitudeTick(const float DeltaTime)
204{
205 if (bIsUpstream || bIsDownstream || !bAltitudeInput)
206 return FVector::ZeroVector;
207
208 if ( MoveComp->MovementMode == MOVE_Flying)
209 {
210 CurrentFlySpeed += Acceleration * DeltaTime;
211 CurrentFlySpeed = FMath::Clamp(CurrentFlySpeed, FlySpeedMin, FlySpeedMax);
212 const FVector Direction = bIsAltitudeUp ? FVector::UpVector : FVector::DownVector;
213
214 if (bIsAltitudeUp)
215 {
216 // 고도 제한
217 if (Owner->GetActorLocation().Z < MaxFlightHeight)
218 return Direction * CurrentFlySpeed * DeltaTime;
219 }
220 else
221 {
222 // 하강일 경우, 지면 근처까지 오면 낙하 중단
223 FVector Start = Owner->GetActorLocation();
224 FVector End = Start - FVector(0.f, 0.f, 88*1.5f); // 캐릭터 크기가 88 이니깐 적당히...1.5밴
225
226 FHitResult HitResult;
227 FCollisionQueryParams Params;
228 Params.AddIgnoredActor(Owner); // 본인 제외
229
230 bool bHit = GetWorld()->LineTraceSingleByChannel(
231 HitResult,
232 Start, End, ECC_Visibility,Params
233 );
234
235 if (bHit)
236 {
237 // 지면에 가까워졌으면 낙하 중단
238 Owner->SetFallingToWalk();
239 return FVector::ZeroVector;
240 }
241
242 return Direction * CurrentFlySpeed * DeltaTime;
243 }
244 }
245
246 return FVector::ZeroVector;
247}
